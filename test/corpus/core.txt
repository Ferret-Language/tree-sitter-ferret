const BuildMode = "debug"

type Point copy struct {
  x i32 = 0
  y i32 = 0
}

fn (p *mut Point) Shift(dx i32, dy i32) Point {
  let mut next: Point = .{ .x = dx, .y = dy }
  return next
}

fn Eval(comptime T Type, value T) T {
  let v = comptime value ?? none
  if value == .{ .x = 1 } {
    return value!! catch value
  } else {
    return &mut value.x
  }
}
