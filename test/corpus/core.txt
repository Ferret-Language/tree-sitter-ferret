fn Run(m Mutex, items [3]i32) i32 {
  let mut sum: i32 = 0

  for items |v| {
    sum = sum + v
  }

  for items |i, v| {
    defer release m
    lock m as g {
      defer release g
      panic "boom"
      continue
    }
  }

  return sum
}
